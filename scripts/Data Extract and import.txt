----Pull the data mappings from an org

cci task run data_mapping_trial --org <CumulusConnectedOrgNameGoesHere>


----Pull the data from the source org into the data.sql in the trialforce directory

cci task run extract_dataset -o mapping datasets/fullTrialData/mapping.yml -o sql_path datasets/fullTrialData/data.sql --org Source_ORG
cci task run extract_dataset -o mapping datasets/fullTrialDataRates/mapping.yml -o sql_path datasets/fullTrialDataRates/data.sql --org Source_ORG




----Load the data manually (You wont need to do this if you are running the scratch org setup script it will do this part)

cci task run load_dataset -o mapping datasets/trialforce/mapping.yml -o sql_path datasets/trialforce/data.sql --org TARGET_ORG
